const int motorPin = 9; // Motor controlled by PWM pin D9

void setup() {
  pinMode(motorPin, OUTPUT); // Set motor pin as output
}

void loop() {
  // Turn motor on at full speed for 3 seconds
  analogWrite(motorPin, 255); 
  delay(3000);
  
  // Slow the motor down to half speed for 3 seconds
  analogWrite(motorPin, 128); 
  delay(3000);
  
  // Turn motor completely off for 2 seconds
  analogWrite(motorPin, 0); 
  delay(2000);
}
