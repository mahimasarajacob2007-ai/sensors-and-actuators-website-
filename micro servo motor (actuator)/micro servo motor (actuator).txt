#include <Servo.h> // Include the built-in Servo library

Servo myServo;  // Create a servo object to control our motor

void setup() {
  myServo.attach(9); // Tell the Arduino the servo signal is on Pin 9
}

void loop() {
  // Sweep from 0 to 180 degrees
  for (int angle = 0; angle <= 180; angle += 1) {
    myServo.write(angle); // Tell servo to go to this position
    delay(15);            // Wait 15ms for the servo to reach it
  }
  
  // Sweep back from 180 to 0 degrees
  for (int angle = 180; angle >= 0; angle -= 1) {
    myServo.write(angle); // Tell servo to go to this position
    delay(15);            // Wait 15ms for the servo to reach it
  }
}
