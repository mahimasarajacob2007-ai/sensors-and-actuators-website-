// --- Pin Assignments ---
const int FSR_PIN = A0;   // Force sensor analog input
const int LED_PIN = 3;    // PWM Digital output to control LED brightness

void setup() {
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);     // Open Serial Monitor to see live pressure numbers
}

void loop() {
  // Read physical squeeze pressure from the FSR sensor (0 to 876)
  int forceValue = analogRead(FSR_PIN);
  
  // Print the raw pressure number to the console screen
  Serial.print("Force Squeeze Value: ");
  Serial.println(forceValue);

  // Map the pressure readings directly to LED brightness (0 = Off, 255 = Full Bright)
  // Tinkercad FSRs typically jump from 0 up to 876 when pressed firmly
  int brightness = map(forceValue, 0, 876, 0, 255);
  
  // Ensure the brightness math stays safely between 0 and 255
  brightness = constrain(brightness, 0, 255);

  // Write the analog PWM dimming signal to the LED
  analogWrite(LED_PIN, brightness);
  
  delay(30); // Small pause to stabilize sensor readings
}
