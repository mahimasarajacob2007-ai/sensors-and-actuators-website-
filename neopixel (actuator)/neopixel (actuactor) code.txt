#include <Adafruit_NeoPixel.h> // Include the built-in NeoPixel library

const int pin = 6;         // Data pin connected to D6
const int numPixels = 10;  // Change this to match the number of LEDs on your strip

// Set up the NeoPixel strip object
Adafruit_NeoPixel strip = Adafruit_NeoPixel(numPixels, pin, NEO_GRB + NEO_KHZ800);

void setup() {
  strip.begin();           // Initialize the strip
  strip.show();            // Turn all pixels off initially
}

void loop() {
  // Loop through every single LED on the strip one by one
  for(int i = 0; i < numPixels; i++) {
    // Set pixel 'i' to Red (Red=255, Green=0, Blue=0)
    strip.setPixelColor(i, strip.Color(255, 0, 0));
    strip.show();          // Refresh the strip to show the change
    delay(100);            // Pause briefly to create a "chase" effect
  }
  
  delay(500); // Wait half a second before changing colors
  
  // Clear the strip and turn them all to Blue
  for(int i = 0; i < numPixels; i++) {
    strip.setPixelColor(i, strip.Color(0, 0, 255)); // Blue
    strip.show();
    delay(100);
  }
  
  delay(500);
}
