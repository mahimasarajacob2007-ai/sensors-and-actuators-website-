const int ldrPin = A0;  // Photoresistor connected to A0
const int ledPin = 13;  // LED connected to D13

void setup() {
  pinMode(ledPin, OUTPUT); // Set LED as output
  Serial.begin(9000);      // Start serial monitor to see values
}

void loop() {
  int ldrValue = analogRead(ldrPin); // Read light levels (0 to 1023)
  Serial.println(ldrValue);          // Print light level to screen

  // If light drops below 400 (it gets dark), turn on the LED
  if (ldrValue < 400) {
    digitalWrite(ledPin, HIGH);
  } else {
    digitalWrite(ledPin, LOW);
  }
  delay(100); 
}
